### Part II: Modifier Mathematics and Interval Representation

**Primary Representation:**
```
⟨[a,b], L, π⟩
```
Where:
- [a,b] = confidence interval for measurement
- L = linguistic qualifier from {certain, probable, possible}
- π = possibility distribution over outcome space

**Modifier Operations:**
```python
def concentration(μ):
    """μ₊(x) = μ(x)² - Increases certainty"""
    return lambda x: μ(x) ** 2

def identity(μ):
    """μ₌(x) = μ(x) - No change"""
    return μ

def dilation(μ):
    """μ₋(x) = √μ(x) - Decreases certainty"""
    return lambda x: np.sqrt(μ(x))
```

```python
def possibility_fusion(report1, report2):
    """Handle conflicting crime reports using possibility theory"""
    
    # Possibility distributions
    π1 = report1['possibility_distribution']
    π2 = report2['possibility_distribution']
    
    # Conjunction (minimum) for agreement
    π_combined = lambda x: min(π1(x), π2(x))
    
    # Disjunction (maximum) for coverage
    π_coverage = lambda x: max(π1(x), π2(x))
    
    # Weighted combination based on source reliability
    α = report1['reliability']
    π_final = lambda x: α * π_combined(x) + (1-α) * π_coverage(x)
    
    return π_final
```

```python
def standardize_coordinates(lat, lon, source_projection=None):
    """Convert all coordinates to WGS84"""
    
    if source_projection == 'state_plane_illinois':
        # Chicago uses State Plane coordinates
        transformer = Transformer.from_crs(
            "EPSG:3435",  # State Plane Illinois East
            "EPSG:4326",  # WGS84
            always_xy=True

        lon, lat = transformer.transform(lon, lat)
    
    # Validate coordinates
    assert -90 <= lat <= 90, f"Invalid latitude: {lat}"
    assert -180 <= lon <= 180, f"Invalid longitude: {lon}"
    
    return round(lat, 6), round(lon, 6)
```

```python
def find_connected_crimes():
    """Comprehensive crime connection discovery"""
    
    # Load enriched crime data
    crimes = load_enriched_crimes()
    
    # Back-azimuth for witness reports
    def calculate_witness_vectors(crime1, crime2):
        """Calculate if crimes could have same witness"""
        
        # Azimuth from crime1 to crime2
        azimuth = calculate_azimuth(
            crime1['lat'], crime1['lon'],
            crime2['lat'], crime2['lon']
        )
        
        # Reverse azimuth
        back_azimuth = (azimuth + 180) % 360
        
        # Time difference
        time_delta = abs(
            crime1['timestamp'] - crime2['timestamp']
        ).total_seconds() / 60  # minutes
        
        # Probability of same witness
        distance = haversine_distance(crime1, crime2)
        if distance < 0.5 and time_delta < 30:  # Within 0.5km and 30min
            return {
                'probability': 0.8,
                'azimuth': azimuth,
                'back_azimuth': back_azimuth
            }
        return None
    
    # Temporal clustering for series
    def detect_crime_series(crimes, time_window=7*24*60):  # 7 days in minutes
        """Identify potential crime series"""
        
        from sklearn.cluster import DBSCAN
        
        # Feature matrix: [time, lat, lon, crime_type]
        features = []
        for crime in crimes:
            features.append([
                crime['timestamp'].timestamp() / 3600,  # Hours
                crime['latitude'] * 100,  # Scale coordinates
                crime['longitude'] * 100,
                hash(crime['primary_type']) % 100  # Crime type encoding
            ])
        
        # DBSCAN clustering
        clusters = DBSCAN(eps=5, min_samples=3).fit(features)
        
        # Group crimes by cluster
        series = {}
        for idx, label in enumerate(clusters.labels_):
            if label != -1:  # Not noise
                if label not in series:
                    series[label] = []
                series[label].append(crimes[idx])
        
        return series
    
    # Modus operandi matching
    def match_modus_operandi(crime1, crime2):
        """Calculate MO similarity score"""
        
        # Extract MO features
        features1 = extract_mo_features(crime1)
        features2 = extract_mo_features(crime2)
        
        # Calculate similarity
        similarity = 0
        weights = {
            'time_of_day': 0.15,
            'day_of_week': 0.10,
            'location_type': 0.20,
            'crime_type': 0.30,
            'weapon': 0.15,
            'victim_profile': 0.10
        }
        
        for feature, weight in weights.items():
            if features1[feature] == features2[feature]:
                similarity += weight
        
        return similarity
    
    # Weapon signature correlation
    def correlate_weapon_signatures(crimes):
        """Identify unique weapon patterns"""
        
        weapon_groups = {}
        
        for crime in crimes:
            if crime.get('weapon_desc'):
                weapon_key = normalize_weapon(crime['weapon_desc'])
                
                if weapon_key not in weapon_groups:
                    weapon_groups[weapon_key] = []
                
                weapon_groups[weapon_key].append(crime)
        
        # Find unusual weapon patterns
        signatures = []
        for weapon, group in weapon_groups.items():
            if len(group) > 2:  # Multiple uses
                # Calculate geographic spread
                lats = [c['latitude'] for c in group]
                lons = [c['longitude'] for c in group]
                
                spread = calculate_geographic_spread(lats, lons)
                
                if spread > 5:  # Spread over 5km
                    signatures.append({
                        'weapon': weapon,
                        'crimes': group,
                        'spread_km': spread,
                        'timespan_days': calculate_timespan(group)
                    })
        
        return signatures
    
    # Execute all connection algorithms
    connections = {
        'witness_connections': [],
        'crime_series': detect_crime_series(crimes),
        'mo_matches': [],
        'weapon_signatures': correlate_weapon_signatures(crimes)
    }
    
    # Find witness connections
    for i, crime1 in enumerate(crimes):
        for crime2 in crimes[i+1:]:
            witness_link = calculate_witness_vectors(crime1, crime2)
            if witness_link:
                connections['witness_connections'].append({
                    'crime1': crime1['id'],
                    'crime2': crime2['id'],

    # Find MO matches
    for i, crime1 in enumerate(crimes):

                connections['mo_matches'].append({
                    'crime1': crime1['id'],
                    'crime2': crime2['id'],
                    'similarity': mo_score
                })
    
    return connections
```

```python
class DroneSecurityNetwork:
    """MMP-based drone surveillance system"""
    
    def __init__(self):

        self.airspace_map = self.load_airspace_classifications()
        self.critical_sites = self.load_critical_infrastructure()
    
    def encode_drone_observation(self, drone_data):
        """Convert drone telemetry to MMP format"""
        
        # Observer component
        observer = {
            'drone_id': drone_data['registration_number'],
            'operator_license': drone_data['operator_license'],
            'timestamp': drone_data['timestamp'],
            'confidence': self.calculate_gps_confidence(drone_data)
        }
        
        # Meta component
        meta = {
            'airspace_class': self.get_airspace_class(
                drone_data['latitude'], 
                drone_data['longitude'],
                drone_data['altitude_ft']
            ),
            'proximity_critical': self.nearest_critical_site(
                drone_data['latitude'],
                drone_data['longitude']
            ),
            'weather_conditions': drone_data.get('weather', 'unknown')
        }
        
        # Kinematic component
        kinematic = {
            'velocity_mps': drone_data['ground_speed'],
            'heading': drone_data['heading'],
            'altitude_change_rate': drone_data['vertical_speed'],
            'flight_pattern': self.classify_flight_pattern(
                drone_data['flight_path']
            )
        }
        
        # Platform component
        platform = {
            'drone_model': drone_data['make_model'],
            'weight_class': self.get_weight_class(drone_data['weight_g']),
            'capabilities': self.infer_capabilities(drone_data['make_model']),
            'payload_type': drone_data.get('declared_payload', 'unknown')
        }
        
        return {
            'observer': observer,
            'meta': meta,
            'kinematic': kinematic,
            'platform': platform
        }
    
    def detect_suspicious_patterns(self, drone_network_data):
        """Identify potential security threats"""
        
        suspicious_patterns = []
        
        for drone_id, trajectory in drone_network_data.items():
            # Check for loitering near critical infrastructure
            loiter_score = self.calculate_loiter_score(trajectory)
            
            # Check for coordinated movements
            coordination_score = self.detect_coordination(
                drone_id, 
                trajectory, 
                drone_network_data
            )
            
            # Check for airspace violations
            violation_score = self.check_airspace_violations(trajectory)
            
            # Aggregate threat score
            threat_score = (
                0.4 * loiter_score +
                0.4 * coordination_score +
                0.2 * violation_score
            )
            
            if threat_score > 0.7:
                suspicious_patterns.append({
                    'drone_id': drone_id,
                    'threat_score': threat_score,
                    'pattern_type': self.classify_threat_pattern(
                        loiter_score, 
                        coordination_score, 
                        violation_score
                    ),
                    'recommended_action': self.recommend_response(threat_score)
                })
        
        return suspicious_patterns
    
    def calculate_loiter_score(self, trajectory):
        """Detect loitering behavior near sensitive sites"""
        
        # Parameters
        loiter_radius = 500  # meters
        loiter_time = 300    # seconds (5 minutes)
        
        # Check each critical site
        max_loiter_score = 0
        
        for site in self.critical_sites:
            time_near_site = 0
            
            for point in trajectory:
                distance = haversine_distance(
                    point['lat'], point['lon'],
                    site['lat'], site['lon']
                )
                
                if distance <= loiter_radius:
                    time_near_site += point['duration']
            
            if time_near_site >= loiter_time:
                # Calculate severity based on site criticality
                loiter_score = min(1.0, time_near_site / loiter_time) * site['criticality']
                max_loiter_score = max(max_loiter_score, loiter_score)
        
        return max_loiter_score
```

```sql
WITH drone_positions AS (
    SELECT 
        drone_id,
        timestamp,
        latitude,
        longitude,
        altitude_ft,
        LAG(latitude) OVER (PARTITION BY drone_id ORDER BY timestamp) as prev_lat,
        LAG(longitude) OVER (PARTITION BY drone_id ORDER BY timestamp) as prev_lon,
        LAG(timestamp) OVER (PARTITION BY drone_id ORDER BY timestamp) as prev_time
    FROM drone_telemetry
    WHERE timestamp >= NOW() - INTERVAL '1 hour'
),
drone_analytics AS (
    SELECT 
        drone_id,
        timestamp,
        latitude,
        longitude,
        altitude_ft,
        -- Calculate speed
        CASE 
            WHEN prev_time IS NOT NULL THEN
                haversine_distance(latitude, longitude, prev_lat, prev_lon) /
                EXTRACT(EPOCH FROM (timestamp - prev_time))
            ELSE 0
        END as speed_mps,
        -- Find nearest critical site
        (SELECT 
            MIN(haversine_distance(latitude, longitude, cs.lat, cs.lon))
         FROM critical_sites cs
        ) as distance_to_critical_m
    FROM drone_positions
)
SELECT 
    drone_id,
    COUNT(*) as suspicious_points,
    AVG(distance_to_critical_m) as avg_distance_critical,
    STDDEV(speed_mps) as speed_variance,
    MIN(altitude_ft) as min_altitude,
    CASE
        WHEN AVG(distance_to_critical_m) < 1000 
             AND STDDEV(speed_mps) < 2 
             AND COUNT(*) > 10 THEN 'HIGH'
        WHEN AVG(distance_to_critical_m) < 2000 
             AND COUNT(*) > 5 THEN 'MEDIUM'
        ELSE 'LOW'
    END as threat_level
FROM drone_analytics
WHERE distance_to_critical_m < 5000  -- Within 5km of critical site
GROUP BY drone_id
HAVING COUNT(*) > 3  -- Multiple observations
ORDER BY threat_level DESC, avg_distance_critical ASC;
```
