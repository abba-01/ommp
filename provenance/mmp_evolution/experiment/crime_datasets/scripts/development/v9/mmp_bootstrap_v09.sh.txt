#!/bin/bash
# mmp_bootstrap_v09.sh - Secure minimal launcher
# Handles STIG security requirements then pulls main installer
set -euo pipefail

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

# Prevent re-runs
SENTINEL="/etc/.mmp_installed"
if [[ -f "$SENTINEL" ]]; then
    echo -e "${RED}Already installed. Remove $SENTINEL to force re-run.${NC}"
    exit 1
fi

echo -e "${BLUE}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
echo -e "${BLUE}           MMP SECURE BOOTSTRAP v09${NC}"
echo -e "${BLUE}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"

# 1. Generate secure credentials
echo -e "\n${YELLOW}[1/5] Generating secure credentials...${NC}"
ADMIN_PASSWORD=$(openssl rand -base64 32 | tr -dc 'A-Za-z0-9!@#$%^&*' | head -c 24)
RECOVERY_PASSPHRASE=$(openssl rand -base64 32 | tr -dc 'A-Za-z0-9!@#$%^&*' | head -c 32)
MYSQL_ROOT_PASS=$(openssl rand -base64 32 | tr -dc 'A-Za-z0-9!@#$%^&*' | head -c 24)
MYSQL_APP_PASS=$(openssl rand -base64 32 | tr -dc 'A-Za-z0-9!@#$%^&*' | head -c 24)
INSTALLATION_ID=$(cat /proc/cpuinfo | grep -E "processor|model" | head -4 | md5sum | cut -c1-8 | tr a-z A-Z)

# 2. Create admin user
echo -e "${YELLOW}[2/5] Creating secure admin user...${NC}"
if ! id maverick &>/dev/null; then
    useradd -m -s /bin/bash maverick
    echo "$ADMIN_PASSWORD" | passwd maverick --stdin &>/dev/null
    usermod -aG wheel maverick
fi

# 3. Store credentials securely
echo -e "${YELLOW}[3/5] Storing encrypted credentials...${NC}"
mkdir -p /home/maverick/.mmp
chmod 700 /home/maverick/.mmp

# Create plaintext for encryption
cat > /tmp/.mmp_creds << EOF
Installation ID: $INSTALLATION_ID
Date: $(date -u)
Admin User: maverick
Admin Pass: $ADMIN_PASSWORD
MySQL Root: $MYSQL_ROOT_PASS
MySQL App: $MYSQL_APP_PASS
Recovery: Use passphrase shown during install
EOF

# Encrypt credentials
openssl enc -aes-256-gcm -pbkdf2 -iter 600000 -salt \
    -in /tmp/.mmp_creds \
    -out /home/maverick/.mmp/credentials.enc \
    -pass pass:"$RECOVERY_PASSPHRASE"

# Secure cleanup
shred -vfz -n 3 /tmp/.mmp_creds
chown -R maverick:maverick /home/maverick/.mmp

# 4. Export vars for main installer
echo -e "${YELLOW}[4/5] Setting environment...${NC}"
export MMP_INSTALLATION_ID="$INSTALLATION_ID"
export MMP_MYSQL_ROOT_PASS="$MYSQL_ROOT_PASS"
export MMP_MYSQL_APP_PASS="$MYSQL_APP_PASS"
export MMP_SECURE_MODE="true"

# 5. Download and run main installer
echo -e "${YELLOW}[5/5] Loading main installer...${NC}"

# Option 1: If main installer is local
MAIN_INSTALLER="./mmp_main_installer_v09.sh"
if [[ -f "$MAIN_INSTALLER" ]]; then
    echo -e "${GREEN}Running main installer...${NC}\n"
    source "$MAIN_INSTALLER"
else
    # Option 2: Download from repo
    echo "Downloading main installer..."
    INSTALLER_URL="https://your-repo.com/mmp_main_installer_v09.sh"
    if command -v curl &>/dev/null; then
        curl -sSL "$INSTALLER_URL" -o /tmp/mmp_main.sh
    else
        wget -q "$INSTALLER_URL" -O /tmp/mmp_main.sh
    fi
    
    if [[ -f /tmp/mmp_main.sh ]]; then
        chmod +x /tmp/mmp_main.sh
        source /tmp/mmp_main.sh
        rm -f /tmp/mmp_main.sh
    else
        echo -e "${RED}Failed to download main installer${NC}"
        exit 1
    fi
fi

# After main installer completes, show credentials ONCE
echo -e "\n${GREEN}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
echo -e "${GREEN}              INSTALLATION COMPLETE${NC}"
echo -e "${GREEN}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
echo -e "\n${RED}⚠️  SAVE THESE CREDENTIALS - SHOWN ONLY ONCE ⚠️${NC}\n"
echo -e "Installation ID: ${YELLOW}$INSTALLATION_ID${NC}"
echo -e "Admin User:      ${YELLOW}maverick${NC}"
echo -e "Admin Password:  ${YELLOW}$ADMIN_PASSWORD${NC}"
echo -e "Recovery Phrase: ${YELLOW}$RECOVERY_PASSPHRASE${NC}"
echo -e "\n${BLUE}To decrypt stored credentials later:${NC}"
echo -e "  sudo -u maverick mmp_decrypt"
echo -e "  [Enter recovery passphrase when prompted]"
echo -e "\n${GREEN}Next steps:${NC}"
echo -e "  1. Save credentials securely"
echo -e "  2. SSH as maverick@$(hostname -I | awk '{print $1}')"
echo -e "  3. Run 'mmp_help' for available commands"

# Mark as complete
touch "$SENTINEL"
echo -e "\n${YELLOW}Installation marked complete.${NC}"
