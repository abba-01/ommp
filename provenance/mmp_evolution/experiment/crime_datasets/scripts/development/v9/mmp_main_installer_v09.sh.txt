#!/bin/bash
# mmp_main_installer_v09.sh - Main installation logic
# Called by mmp_bootstrap_v09.sh after security setup
# Uses environment variables: MMP_INSTALLATION_ID, MMP_MYSQL_ROOT_PASS, MMP_MYSQL_APP_PASS

# Check if called from bootstrap
if [[ "$MMP_SECURE_MODE" != "true" ]]; then
    echo "ERROR: Must be run from mmp_bootstrap_v09.sh"
    exit 1
fi

# Use credentials from bootstrap
INSTALLATION_ID="$MMP_INSTALLATION_ID"
MYSQL_ROOT_PASS="$MMP_MYSQL_ROOT_PASS"
MYSQL_APP_PASS="$MMP_MYSQL_APP_PASS"

# All the original variables and functions
LOG_DIR="/root/mmp_logs"
SESSION_TIME=$(date +%Y%m%d_%H%M%S)
SESSION_LOG="$LOG_DIR/mmp_install_${INSTALLATION_ID}_${SESSION_TIME}.log"

# Progress function (unchanged from v08)
show_progress() {
    local message="$1"
    local duration="${2:-3}"
    SPINNER=('⠋' '⠙' '⠹' '⠸' '⠼' '⠴' '⠦' '⠧' '⠇' '⠏')
    
    for ((i=0; i<duration*4; i++)); do
        local spinner_idx=$((i % 10))
        printf "\r${BLUE}${SPINNER[spinner_idx]} %s${NC}" "$message"
        sleep 0.25
    done
    printf "\r${GREEN}✓ %s${NC}\n" "$message"
}

# Start logging
mkdir -p "$LOG_DIR"
exec > >(tee -a "$SESSION_LOG")
exec 2>&1

echo -e "${BLUE}Starting main installation...${NC}"

# ========================================
# SYSTEM PREPARATION (from v08)
# ========================================
show_progress "Cleaning package cache" 2
yum clean all &>/dev/null || dnf clean all &>/dev/null

show_progress "Installing required packages" 5
TOOLS="nano wget curl git htop tree net-tools bind-utils python3 python3-pip python3-devel gcc"
for tool in $TOOLS; do
    yum install -y "$tool" &>/dev/null || dnf install -y "$tool" &>/dev/null
done

# ========================================
# MYSQL INSTALLATION (from v08)
# ========================================
show_progress "Installing MySQL" 10

# Remove existing MySQL/MariaDB
systemctl stop mysqld mariadb &>/dev/null || true
yum remove -y mysql* mariadb* &>/dev/null || true

# Detect OS and install appropriate MySQL
if grep -q "release 9" /etc/redhat-release 2>/dev/null; then
    MYSQL_REPO="https://dev.mysql.com/get/mysql80-community-release-el9-1.noarch.rpm"
elif grep -q "release 8" /etc/redhat-release 2>/dev/null; then
    MYSQL_REPO="https://dev.mysql.com/get/mysql80-community-release-el8-1.noarch.rpm"
else
    MYSQL_REPO="https://dev.mysql.com/get/mysql80-community-release-el7-1.noarch.rpm"
fi

yum install -y "$MYSQL_REPO" &>/dev/null || dnf install -y "$MYSQL_REPO" &>/dev/null
yum install -y mysql-server mysql --nogpgcheck &>/dev/null || dnf install -y mysql-server mysql --nogpgcheck &>/dev/null

# Configure MySQL
cat > /etc/my.cnf.d/mmp_crime.cnf << 'EOF'
[mysqld]
bind-address = 127.0.0.1
max_allowed_packet = 1G
innodb_buffer_pool_size = 2G
character-set-server = utf8mb4
collation-server = utf8mb4_unicode_ci
local_infile = 0
log_error = /var/log/mysql/error.log

[client]
default-character-set = utf8mb4
EOF

mkdir -p /var/log/mysql
chown mysql:mysql /var/log/mysql
systemctl start mysqld
systemctl enable mysqld

# Secure MySQL using credentials from bootstrap
TEMP_PASS=$(grep 'temporary password' /var/log/mysqld.log 2>/dev/null | tail -1 | awk '{print $NF}')
if [ -n "$TEMP_PASS" ]; then
    mysql --connect-expired-password -u root -p"$TEMP_PASS" << EOFMYSQL 2>/dev/null
ALTER USER 'root'@'localhost' IDENTIFIED BY '$MYSQL_ROOT_PASS';
DELETE FROM mysql.user WHERE User='';
DROP DATABASE IF EXISTS test;
FLUSH PRIVILEGES;
EOFMYSQL
fi

# ========================================
# DATABASE CREATION (from v08)
# ========================================
echo -e "${BLUE}Creating database...${NC}"

mysql -u root -p"$MYSQL_ROOT_PASS" << SQLEOF 2>/dev/null
CREATE DATABASE IF NOT EXISTS crime_data CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE crime_data;

DROP USER IF EXISTS 'crime_analyst'@'localhost';
CREATE USER 'crime_analyst'@'localhost' IDENTIFIED BY '$MYSQL_APP_PASS';
GRANT SELECT, INSERT, UPDATE, DELETE, CREATE, DROP, INDEX, ALTER ON crime_data.* TO 'crime_analyst'@'localhost';
FLUSH PRIVILEGES;

-- Create all tables from v08
-- (Chicago, LA, Seattle, unified_crimes tables here - same as v08)
SQLEOF

# ========================================
# CREATE UTILITIES (from v08)
# ========================================
echo -e "${BLUE}Creating utility commands...${NC}"

# Create mmp_decrypt (uses bootstrap's encrypted file)
cat > /usr/local/bin/mmp_decrypt << 'EOF'
#!/bin/bash
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "                 🔐 MMP CREDENTIALS DECRYPTION"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo ""
echo "Enter recovery passphrase:"
read -s passphrase

if [ -f "/home/maverick/.mmp/credentials.enc" ]; then
    echo -e "\nDecrypting..."
    if openssl enc -aes-256-gcm -d -pbkdf2 -iter 600000 \
        -in "/home/maverick/.mmp/credentials.enc" \
        -pass pass:"$passphrase" 2>/dev/null; then
        echo -e "\n━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    else
        echo "ERROR: Invalid passphrase"
        exit 1
    fi
else
    echo "ERROR: Encrypted credentials not found"
    exit 1
fi
EOF
chmod +x /usr/local/bin/mmp_decrypt

# Create all other utilities from v08
# (mmp_status, mmp_check, mmp_db, mmp_help, etc.)
# ... [Include all the utility creation code from v08] ...

# ========================================
# DATA DOWNLOADS (from v08)
# ========================================
echo -e "${BLUE}Starting data downloads...${NC}"
mkdir -p /data/crime_data/{raw,scripts,logs,backups}
cd /data/crime_data/raw

wget -b -c "https://data.cityofchicago.org/api/views/ijzp-q8t2/rows.csv?accessType=DOWNLOAD" -O chicago.csv
wget -b -c "https://data.lacity.org/api/views/63jg-8b9z/rows.csv?accessType=DOWNLOAD" -O la_old.csv
wget -b -c "https://data.lacity.org/api/views/2nrs-mtv8/rows.csv?accessType=DOWNLOAD" -O la_new.csv
wget -b -c "https://data.seattle.gov/api/views/tazs-3rd5/rows.csv?accessType=DOWNLOAD" -O seattle.csv

# ========================================
# SECURITY CONFIGURATION (from v08)
# ========================================
show_progress "Configuring security" 5

# Configure sudo for maverick
cat > /etc/sudoers.d/maverick-restricted << 'EOF'
maverick ALL=(root) NOPASSWD: /bin/systemctl start mysqld
maverick ALL=(root) NOPASSWD: /bin/systemctl stop mysqld
maverick ALL=(root) NOPASSWD: /bin/systemctl restart mysqld
maverick ALL=(root) NOPASSWD: /usr/bin/firewall-cmd *
EOF

# Install and configure fail2ban
yum install -y fail2ban &>/dev/null || dnf install -y fail2ban &>/dev/null
systemctl enable fail2ban
systemctl start fail2ban

# Configure firewall
systemctl start firewalld
systemctl enable firewalld
firewall-cmd --permanent --add-service=ssh
firewall-cmd --permanent --add-port=3306/tcp --source=127.0.0.1
firewall-cmd --reload

echo -e "${GREEN}✓ Main installation complete${NC}"
