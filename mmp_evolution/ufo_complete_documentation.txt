@app.route('/')
def dashboard():
    return render_template_string(DASHBOARD_HTML)

@app.route('/api/stats')
def stats():
    conn = mysql.connector.connect(**DB_CONFIG)
    cursor = conn.cursor(dictionary=True)
    
    # Summary stats
    cursor.execute("""
        SELECT 'Total Crimes' as metric, FORMAT(SUM(crime_count), 0) as value
        FROM crime_summary_stats
        UNION ALL
        SELECT 'Active Years', CONCAT(MIN(year), '-', MAX(year))
        FROM crime_summary_stats
        UNION ALL
        SELECT 'Cities Tracked', COUNT(DISTINCT city)
        FROM crime_summary_stats
    """)
    summary = cursor.fetchall()
    
    # Monthly trends (last 12 months)
    cursor.execute("""
        SELECT 
            city,
            CONCAT(year, '-', LPAD(month, 2, '0')) as month,
            SUM(crime_count) as crimes
        FROM crime_summary_stats
        WHERE CONCAT(year, '-', LPAD(month, 2, '0')) >= DATE_FORMAT(DATE_SUB(NOW(), INTERVAL 12 MONTH), '%Y-%m')
        GROUP BY city, year, month
        ORDER BY year, month
    """)
    trends = cursor.fetchall()
    
    # City totals
    cursor.execute("""
        SELECT city, SUM(crime_count) as total
        FROM crime_summary_stats
        GROUP BY city
    """)
    city_totals = cursor.fetchall()
    
    # Get unique cities
    cities = list(set(t['city'] for t in trends))
    
    cursor.close()
    conn.close()
    
    return jsonify({
        'summary': summary,
        'trends': trends,
        'city_totals': city_totals,
        'cities': cities
    })

if __name__ == '__main__':
    app.run(host='127.0.0.1', port=5000, debug=False)
