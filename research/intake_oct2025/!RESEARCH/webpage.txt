        <p>The complete MMP classification space is defined as:</p>

            <li>B ∈ {E, M, V, L, P, J, S, U, N, I} (Body prefixes)</li>
            <li>P ∈ {∅, -, +} × {g, f, s, o, l, d} × ℝ (Position system)</li>
            <li>M ∈ {T, A, L, O, I, S, G, U} (Meta domains)</li>
            <li>K ∈ {Z, L, C, E, R, P, O, U} (Kinematic states)</li>
            <li>S ∈ {T, V, R, C, F, S, O, D, P, L, H, A, E, U} (Spatial morphologies)</li>

            \[\sigma_{shape} = \sigma_0 + k \times \rho(h) \times d^{1.2}\]
        </div>
        
        <p>Where ρ(h) is atmospheric density and d is distance to object.</p>

        <h3>7.1 Field-Ready Classification Algorithm</h3>
        <div class="code-block">
def generate_mmp_code(observation):
    # Determine body prefix
    body = observation.celestial_body  # E, M, V, L, P, etc.
    
    # Classify observer position
    position = observation.observer_position  # g, f, s, o, l, d
    altitude = observation.altitude
    
    # Determine position modifier
    modifier = get_position_modifier(position, altitude)
    
    # Add specificity if needed
    if observation.requires_precision:
        specificity = str(int(altitude))
    else:
        specificity = ""
    
    # Core MMP classification
    domain = classify_domain(body, position, altitude)
    kinematics = analyze_motion(observation.trajectory)
    morphology = identify_shape(observation.imagery)
    
    # Construct complete code
    code = f"{body}{modifier}{position}{specificity}{domain}{kinematics}{morphology}"
    
    return code

def get_position_modifier(position, altitude):
    standards = {
        'g': (0, 0),          # Ground level
        'f': (30000, 45000),  # Flight corridor (feet)
        's': (0, 100),        # Submersion (meters)
        'o': (200, 500),      # LEO (km)
    }
    
    if position in standards:
        min_alt, max_alt = standards[position]
        if altitude < min_alt:
            return '-'
        elif altitude > max_alt:
            return '+'
    return ''  # Standard range
        </div>

CREATE TABLE observations (
    id BIGSERIAL PRIMARY KEY,
    -- Position components
    body_prefix CHAR(1) NOT NULL,
    position_modifier CHAR(1),
    observer_position CHAR(1) NOT NULL,
    altitude_specificity INTEGER,
    
    -- Core MMP components  
    domain CHAR(1) NOT NULL,
    kinematics CHAR(1) NOT NULL,
    morphology CHAR(1) NOT NULL,
    
    -- Metadata
    timestamp TIMESTAMP WITH TIME ZONE,
    location GEOGRAPHY(POINT, 4326),
    confidence DECIMAL(3,2),
    
    -- Computed fields
    mmp_code VARCHAR(20) GENERATED ALWAYS AS (
        body_prefix || 
        COALESCE(position_modifier, '') || 
        observer_position || 
        COALESCE(altitude_specificity::TEXT, '') ||
        domain || kinematics || morphology
    ) STORED,
    
    -- Indexes
    INDEX idx_body (body_prefix),
    INDEX idx_position (observer_position, altitude_specificity),
    INDEX idx_mmp_code (mmp_code),
    INDEX idx_classification (domain, kinematics, morphology)
);
