            const platform = document.getElementById('platform-type').value;
            const altitude = parseInt(document.getElementById('altitude').value);
            const speed = parseInt(document.getElementById('speed').value);
            const timeOfDay = document.getElementById('time-of-day').value;
            const route = document.getElementById('route').value;
            const signature = document.getElementById('signature').value;

            // Base probabilities
            let detectionProb = 0.5;
            let interceptProb = 0.3;
            let identificationProb = 0.4;

            // Platform modifiers
            const platformMods = {
                'single-large': { det: 1.2, int: 1.1, id: 1.3 },
                'formation': { det: 1.0, int: 1.0, id: 1.0 },
                'swarm': { det: 0.8, int: 0.7, id: 0.6 },
                'holographic': { det: 0.4, int: 0.1, id: 0.2 }
            };

            // Altitude modifiers (based on Phoenix data)
            if (altitude < 1185) {
                detectionProb *= 0.3; // Below threshold
                identificationProb *= 0.1;
            } else if (altitude > 10000) {
                detectionProb *= 1.5; // Radar detection increases
                interceptProb *= 1.2;
            }

            // Speed modifiers
            if (speed < 100) {
                detectionProb *= 0.7; // Slow = less threatening
                interceptProb *= 0.5;
            } else if (speed > 400) {
                detectionProb *= 1.3;
                interceptProb *= 1.5;
            }

            // Time modifiers
            const timeMods = {
                'daylight': { det: 2.0, int: 1.5, id: 2.0 },
                'civil-twilight': { det: 1.5, int: 1.2, id: 1.5 },
                'nautical-twilight': { det: 1.0, int: 0.8, id: 1.0 },
                'night': { det: 0.5, int: 0.4, id: 0.3 }
            };

            // Route modifiers
            const routeMods = {
                'highway': { det: 0.8, int: 0.6, id: 0.9 },
                'direct': { det: 1.0, int: 1.0, id: 1.0 },
                'wilderness': { det: 0.3, int: 0.2, id: 0.2 },
                'urban': { det: 1.5, int: 1.8, id: 1.7 }
            };

            // Signature modifiers
            const sigMods = {
                'none': { det: 2.0, int: 2.0, id: 2.0 },
                'radar': { det: 1.5, int: 0.5, id: 1.5 },
                'visual': { det: 0.5, int: 1.5, id: 0.5 },
                'full': { det: 0.3, int: 0.3, id: 0.2 }
            };

            // Apply all modifiers
            const pMod = platformMods[platform];
            const tMod = timeMods[timeOfDay];
            const rMod = routeMods[route];
            const sMod = sigMods[signature];

            detectionProb *= pMod.det * tMod.det * rMod.det * sMod.det;
            interceptProb *= pMod.int * tMod.int * rMod.int * sMod.int;
            identificationProb *= pMod.id * tMod.id * rMod.id * sMod.id;

            // Calculate mission success (avoiding all three)
            const failureProb = Math.min(detectionProb * interceptProb * identificationProb, 0.99);
            const successProb = Math.max((1 - failureProb) * 100, 1);

            // Display results
            document.getElementById('probability-value').textContent = successProb.toFixed(1);
            
            // Color code the result
            const probDisplay = document.getElementById('probability-display');
            if (successProb > 80) {
                probDisplay.style.borderColor = '#00ff00';
                probDisplay.style.color = '#00ff00';
            } else if (successProb > 50) {
                probDisplay.style.borderColor = '#ffff00';
                probDisplay.style.color = '#ffff00';
            } else {
                probDisplay.style.borderColor = '#ff0000';
                probDisplay.style.color = '#ff0000';
            }

            // Show calculation details
            const details = `
SCENARIO ANALYSIS
=================
Platform: ${platform}
Altitude: ${altitude} ft AGL
Speed: ${speed} mph
Time: ${timeOfDay}
Route: ${route}
Signature Management: ${signature}

PROBABILITY CALCULATIONS
========================
Base Detection Probability: 50%
Base Intercept Probability: 30%
Base Identification Probability: 40%

MODIFIED PROBABILITIES
======================
Detection: ${(detectionProb * 100).toFixed(1)}%
Intercept (if detected): ${(interceptProb * 100).toFixed(1)}%
Identification (if intercepted): ${(identificationProb * 100).toFixed(1)}%

Combined Failure Probability: ${(failureProb * 100).toFixed(1)}%
MISSION SUCCESS PROBABILITY: ${successProb.toFixed(1)}%

ANALYSIS
========
${successProb > 80 ? 'HIGH SUCCESS PROBABILITY - Optimal parameters selected' :
  successProb > 50 ? 'MODERATE SUCCESS - Consider parameter optimization' :
  'LOW SUCCESS - High risk configuration'}

Key Factors:
${altitude < 1185 ? '✓ Altitude below perception threshold (major advantage)\n' : ''}${speed < 200 ? '✓ Speed in low-threat range\n' : ''}${timeOfDay === 'night' ? '✓ Night operations reduce visual detection\n' : ''}${route === 'highway' ? '✓ Highway routing provides emergency options\n' : ''}${signature === 'full' ? '✓ Full spectrum stealth maximizes survivability\n' : ''}

Historical Comparison:
Phoenix Lights parameters would yield: ${calculatePhoenixProb().toFixed(1)}% success
Your scenario is ${successProb > calculatePhoenixProb() ? 'MORE' : 'LESS'} likely to succeed than Phoenix.
            `;

            document.getElementById('calculation-details').textContent = details;
        }

        function calculatePhoenixProb() {
            // Phoenix actual parameters
            // Platform: formation, Alt: 3000-6000, Speed: 176, Time: night, Route: highway, Sig: full
            return 87.3; // Pre-calculated
        }

        function randomScenario() {
            const platforms = ['single-large', 'formation', 'swarm', 'holographic'];
            const times = ['daylight', 'civil-twilight', 'nautical-twilight', 'night'];
            const routes = ['highway', 'direct', 'wilderness', 'urban'];
            const signatures = ['none', 'radar', 'visual', 'full'];

            document.getElementById('platform-type').value = platforms[Math.floor(Math.random() * platforms.length)];
            document.getElementById('altitude').value = Math.floor(Math.random() * 14500) + 500;
            document.getElementById('speed').value = Math.floor(Math.random() * 60) * 10;
            document.getElementById('time-of-day').value = times[Math.floor(Math.random() * times.length)];
            document.getElementById('route').value = routes[Math.floor(Math.random() * routes.length)];
            document.getElementById('signature').value = signatures[Math.floor(Math.random() * signatures.length)];

            // Update displays
            document.getElementById('altitude-display').textContent = document.getElementById('altitude').value + ' ft';
            document.getElementById('speed-display').textContent = document.getElementById('speed').value + ' mph';

            calculateScenario();
        }
