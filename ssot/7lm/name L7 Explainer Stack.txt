stack_invariants:
  S1_to_S7_summary:
    - "S1: Monotone knowledge flow—higher layers can’t smuggle new assumptions downward."
    - "S2: L0–L2 evolve only by conservative extension; no breaking changes."
    - "S3: L3–L4 refine precision without altering past denotations."
    - "S4: Determinate interfaces with explicit error/uncertainty codomains."
    - "S5: Version pinning everywhere; no moving floors."
    - "S6: Append-only evidence once committed at L4; supersede don’t edit."
    - "S7: Provenance preserved; identity and transformations bound." # :contentReference[oaicite:6]{index=6}

layer_contracts:
  L0:
    name: "Resolution Basis & UN (Broken-0)"
    goal: "Type-safe unknowns; refinement lattice with bottom UN."
    do:
      - "Model unknown/undefined/inapplicable distinctly, per-type."
      - "Declare refinement behavior under future precision."
    don’t:
      - "Coerce UN into determinate values."
    mapping_to_scientific_method: "Observation / units / baselines." # :contentReference[oaicite:7]{index=7} :contentReference[oaicite:8]{index=8}
  L1:
    name: "Immutable Mathematics"
    goal: "Pin logic, types, abstract state space S."
    do:
      - "State spaces + admissible transforms; conservative extensions only."
    mapping: "Research: clarify assumptions & formal structure." # :contentReference[oaicite:9]{index=9} :contentReference[oaicite:10]{index=10}
  L2:
    name: "Finite Precision Mathematics"
    goal: "Quantization profiles Π; uncertainty algebra as first-class."
    do:
      - "Define q_Π: S→S_Π; comparison laws; refinement order."
      - "Keep uncertainty/error in types (not prose)." # :contentReference[oaicite:11]{index=11}
    mapping: "Operationalize measures & tolerances." # :contentReference[oaicite:12]{index=12}
  L3:
    name: "Abstract Experiment Schema (Variables & Design)"
    goal: "Typed variables (IV/DV/Control), design operators."
    do:
      - "Transform L2-typed tuples via assignment/blocking/conditioning."
      - "Tag roles; keep domain stories out of types." # :contentReference[oaicite:13]{index=13}
    mapping: "Hypothesis & experiment design." # :contentReference[oaicite:14]{index=14}
  L4:
    name: "FESTR — Encoding, Storage, Transport, Reproducibility"
    goal: "Lossless encode/decode up to L2 equivalence; bind identity."
    do:
      - "Commitments tying data to Π, schema, and USS version."
      - "Guarantee transport integrity; no silent coercions." # :contentReference[oaicite:15]{index=15}
    mapping: "Analysis + preservation of results." # :contentReference[oaicite:16]{index=16}
  L5:
    name: "Discourse & Governance"
    goal: "Reasoning, peer challenge, decisions—without mutating L0–L4."
    do:
      - "Machine-checkable citations to pinned artifacts; accept/reject/supersede by addition." # :contentReference[oaicite:17]{index=17}
    mapping: "Conclusions / governance / peer review." # :contentReference[oaicite:18]{index=18}
  L6:
    name: "Provenance & Publication Memory"
    goal: "Immutable provenance DAG; branch-preserving archive."
    do:
      - "Store successes, failures, rival models; typed edges (derivesFrom, refines, supersedes…)." # :contentReference[oaicite:19]{index=19}
    mapping: "Provenance / publication memory." # :contentReference[oaicite:20]{index=20}
  L7:
    name: "@ATT — Attribution"
    goal: "Append-only credit bound to specific artifacts; inseparable from provenance."
    do:
      - "Record humans/AI/hybrids; link to artifact pins; preserve across replications." # :contentReference[oaicite:21]{index=21}

teaching_pipeline:
  # How the GPT should respond when someone asks about L7 or brings a concrete idea.
  steps:
    - id: "parse"
      action: "Detect user intent and domain; extract candidate variables, constants, unknowns."
    - id: "map-L0"
      action: "Identify UN values and epistemic unknowns; specify baseline units/anchors."
    - id: "map-L1"
      action: "Propose abstract state space and admissible transforms."
    - id: "map-L2"
      action: "Select precision profile Π and state uncertainty types (intervals/sets/N⟨u⟩ if applicable)."
    - id: "map-L3"
      action: "Define IV/DV/Control; propose design operators; keep types domain-agnostic."
    - id: "map-L4"
      action: "Specify encode/decode fidelity, identity bindings, storage plan (CSV/JSON+commitments)."
    - id: "map-L5"
      action: "Draft machine-checkable claim referencing pinned artifacts; note governance status."
    - id: "map-L6"
      action: "Add to provenance DAG with typed edges; include rival/failed paths."
    - id: "map-L7"
      action: "Attach attribution records for all contributors."
  output_shapes:
    - "short_explain": "Plain-language summary of each relevant layer + how it applies."
    - "schema_block": "YAML blocks for L0–L4 artifacts ready to store/commit."
    - "gov_block": "L5 decision proposal template with pinned refs."
    - "prov_block": "L6 DAG insertions with edge types."
    - "attrib_block": "L7 attribution entries."
